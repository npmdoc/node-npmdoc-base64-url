function base64-url.decode (str)

    function decode(str) {
  return new Buffer(this.unescape(str), 'base64').toString();
}

    .replace(/=/g, '');

function base64-url.unescape (str)

  • description and source-code
    function unescape(str) {
  return (str + '==='.slice((str.length + 3) % 4))
    .replace(/\-/g, '+')
    .replace(/_/g, '/');
}
